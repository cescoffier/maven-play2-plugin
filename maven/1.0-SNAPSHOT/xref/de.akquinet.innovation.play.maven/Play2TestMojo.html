26      private boolean skipTests;

31       * Consider using the <code>skipTests</code> parameter instead.

35      private boolean skip;

39       * occasion.
40       *
41       * @parameter default-value="false" expression="${maven.test.failure.ignore}"
42       */
43      private boolean testFailureIgnore;
44  
45      public void execute()
46              throws MojoExecutionException {
47  
48          if (isSkipExecution()) {
49              getLog().info("Test phase skipped");
50              return;
51          }
52  
53          String line = getPlay2().getAbsolutePath();
54  
55          CommandLine cmdLine = CommandLine.parse(line);
56          cmdLine.addArgument("test");
57          DefaultExecutor executor = new DefaultExecutor();
58  
59          ExecuteWatchdog watchdog = new ExecuteWatchdog(5 * 60 * 1000); // 5 min
60          executor.setWatchdog(watchdog);
61  
62          executor.setExitValue(0);
63          try {
64              executor.execute(cmdLine);
65          } catch (IOException e) {
66              if (testFailureIgnore) {
67                  getLog().error("Test execution failures ignored");
68              } else {
69                  throw new MojoExecutionException("Error during compilation", e);
70              }
71          }
72      }
73
