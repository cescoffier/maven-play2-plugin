28  /**
29   * Checks the common methods of Play 2 mojos
30   */
31  public class Play2MojoTest {
32  
33      private String old_play2_system_var;
34  
35      @Before
36      public void setUp() {
37          // Store the current configuration
38          old_play2_system_var = System.getProperty(AbstractPlay2Mojo.ENV_PLAY2_HOME);
39      }
40  
41      @After
42      public void tearDown() {
43          // Restore configuration
44          System.clearProperty(AbstractPlay2Mojo.ENV_PLAY2_HOME);
45          if (old_play2_system_var != null) {
46              System.setProperty(AbstractPlay2Mojo.ENV_PLAY2_HOME, old_play2_system_var);
47          }
48      }
49  
50      @Test
51      public void testPlay2LocationUsingSystemVariable() throws MojoExecutionException {
52          System.setProperty(AbstractPlay2Mojo.ENV_PLAY2_HOME, "somewhere");
53          Play2CleanMojo mojo = new Play2CleanMojo();
54          mojo.play2Home = "ignored";
55          assertThat(mojo.getPlay2Home()).isEqualTo("somewhere");
56      }
57  
58      @Test
59      public void testPlay2LocationUsingSetting() throws MojoExecutionException {
60          System.clearProperty(AbstractPlay2Mojo.ENV_PLAY2_HOME);
61  
62          // I don't really like this conditional test, but I can't be usre of the existence of the env variable.
63          Play2CleanMojo mojo = new Play2CleanMojo();
64          mojo.play2Home = "somewhere";
65  
66          assertThat(mojo.getPlay2Home()).isEqualTo("somewhere");
67      }
68  
69      @Test
70      public void testPlay2LocationUsingEnvVariable() throws MojoExecutionException {
71          String env = System.getenv(AbstractPlay2Mojo.ENV_PLAY2_HOME);
72          System.clearProperty(AbstractPlay2Mojo.ENV_PLAY2_HOME);
73  
74          // I don't really like this conditional test, but I can't be sure of the existence of the env variable.
75          Play2CleanMojo mojo = new Play2CleanMojo();
76          if (env == null) {
77              mojo.play2Home = "somewhere";
78              env = "somewhere";
79          }
80  
81          assertThat(mojo.getPlay2Home()).isEqualTo(env);
82      }
83  
84      @Test(expected = MojoExecutionException.class)
85      public void testTimeout() throws MojoExecutionException {
86          System.clearProperty(AbstractPlay2Mojo.ENV_PLAY2_HOME);
87          Play2CleanMojo mojo = new Play2CleanMojo();
88          mojo.timeout = 1; // 1 ms is just not enough
89          mojo.project = mock(MavenProject.class);
90          when(mojo.project.getArtifactId()).thenReturn("my-artifact-id");
91          when(mojo.project.getGroupId()).thenReturn("my-group-id");
92          when(mojo.project.getVersion()).thenReturn("0.0.1");
93  
94          mojo.execute();
95      }
