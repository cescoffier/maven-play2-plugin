18  import org.apache.maven.project.MavenProject;
19  import org.junit.Test;
20  import org.mockito.Mockito;
21  
22  import static org.fest.assertions.Assertions.assertThat;
23  
24  /**
25   * Checks the behavior of the Update SBT Build Mojo.
26   */
27  public class UpdateSBTBuildMojoTest {
28  
29      @Test
30      public void testAppNameReplacement() {
31          UpdateSBTBuildMojo mojo = new UpdateSBTBuildMojo();
32  
33          MavenProject project = Mockito.mock(MavenProject.class);
34          Mockito.when(project.getArtifactId()).thenReturn("artifactId");
35  
36          mojo.project = project;
37  
38          // Line that should not be modified.
39          String line = "val appName         = \"artifactId\"";
40          String updated = mojo.manageAppName(line);
41          assertThat(updated).isEqualTo(line);
42  
43          // Empty line
44          line = "val appName         = \"\"";
45          updated = mojo.manageAppName(line);
46          assertThat(updated).isNotEqualTo(line);
47          assertThat(updated).contains("artifactId");
48  
49          // Must be sync
50          line = "val appName         = \"not the same\"";
51          updated = mojo.manageAppName(line);
52          assertThat(updated).isNotEqualTo(line);
53          assertThat(updated).contains("artifactId");
54  
55          // No match
56          line = "val stuff         = \"something\"";
57          updated = mojo.manageAppName(line);
58          assertThat(updated).isEqualTo(line);
59      }
60  
61      @Test
62      public void testVersionReplacement() {
63          UpdateSBTBuildMojo mojo = new UpdateSBTBuildMojo();
64  
65          MavenProject project = Mockito.mock(MavenProject.class);
66          Mockito.when(project.getVersion()).thenReturn("1.0.0-SNAPSHOT");
67  
68          mojo.project = project;
69  
70          // Line that should not be modified.
71          String line = "val appVersion         = \"1.0.0-SNAPSHOT\"";
72          String updated = mojo.manageVersion(line);
73          assertThat(updated).isEqualTo(line);
74  
75          // Empty line
76          line = "val appVersion         = \"\"";
77          updated = mojo.manageVersion(line);
78          assertThat(updated).isNotEqualTo(line);
79          assertThat(updated).contains("1.0.0-SNAPSHOT");
80  
81          // Must be sync
82          line = "val appVersion         = \"not the same\"";
83          updated = mojo.manageVersion(line);
84          assertThat(updated).isNotEqualTo(line);
85          assertThat(updated).contains("1.0.0-SNAPSHOT");
86  
87          // No match
88          line = "val stuff         = \"something\"";
