39   * Package the Play application.
40   * The application is packaged as a Jar file. It is also possible to create the distribution package (zip).

42   * @goal package
43   * @phase package

100     public void execute()
101             throws MojoExecutionException {
102 
103         // Package
104         packageApplication();
105         File packagedApplication = moveApplicationPackageToTarget();
106 
107         // Distribution
108         File dist = null;
109         if (buildDist) {
110             packageDistribution();
111             dist = moveDistributionArtifactToTarget();
112 
113             // The javadoc and source files are created during the distribution construction.
114             moveJavadocAndSourcesArtifactsToTarget();
115 
116             if (!additionalFiles.isEmpty()) {
117                 packageAdditionalFiles(additionalFiles, dist);
118             }
119         }
120         attachArtifactsToProject(packagedApplication, dist);
121     }

123     private void packageAdditionalFiles(List<String> additionalFiles, File distributionFile) throws MojoExecutionException {
124         try {
125             ZipFile zipFile = new ZipFile(distributionFile);
126 
127             ArrayList<File> filesToAdd = new ArrayList<File>(additionalFiles.size());
128             for (String file : additionalFiles) {
129                 File fileToAdd = new File(file);
130                 if (!fileToAdd.exists()) {
131                     throw new MojoExecutionException(fileToAdd.getCanonicalPath() + " not found, can't add to package");
132                 }
133                 filesToAdd.add(fileToAdd);
134 
135                 ZipParameters parameters = new ZipParameters();
136                 parameters.setCompressionLevel(Zip4jConstants.DEFLATE_LEVEL_FASTEST);
137                 parameters.setIncludeRootFolder(true);
138                 // Let's safely assume that the zip filename is also the root directory all files are packaged in
139                 parameters.setRootFolderInZip(StringUtils.substringBeforeLast(distributionFile.getName(), ".zip"));
140                 parameters.setReadHiddenFiles(true);
141 
142                 String message = String.format("Adding files to distribution zip [%s]: \n\t%s",
143                         distributionFile.getCanonicalPath(), StringUtils.join(additionalFiles, "\n\t"));
144                 getLog().info(message);
145 
146                 zipFile.addFiles(filesToAdd, parameters);
147             }
148         } catch (ZipException e) {
149             throw new MojoExecutionException("Cannot add files to zipfile: " + distributionFile, e);
150         } catch (IOException e) {
151             throw new MojoExecutionException("Cannot add files to zipfile: " + distributionFile, e);
152         }
153     }

155     private File moveApplicationPackageToTarget() throws MojoExecutionException {
156         File target = getBuildDirectory();
157         File[] files = FileUtils.convertFileCollectionToFileArray(
158                 FileUtils.listFiles(target, new PackageFileFilter(), new PrefixFileFilter("scala-")));
159 
160         File mainJar = null;
161         for (File file : files) {
162             // We need to distinguish which file is the main artifacts.
163             // We filter out -sources and -javadoc files.
164             if (! file.getName().endsWith("-javadoc.jar")
165                     && ! file.getName().endsWith("-sources.jar")) {
166                 mainJar = file;
167             }
168         }
169 
170         if (mainJar == null) {
171             throw new MojoExecutionException("Cannot find packaged file");
172         }
173 
174         try {
175             if (StringUtils.isBlank(classifier)) {
176                 File out = new File(target, project.getBuild().getFinalName() + ".jar");
177                 getLog().info("Copying " + files[0].getName() + " to " + out.getName());
178                 FileUtils.copyFile(mainJar, out);
179                 return out;
180             } else {
181                 File out = new File(target, project.getBuild().getFinalName() + "-" + classifier + ".jar");
182                 getLog().info("Copying " + files[0].getName() + " to " + out.getName());
183                 FileUtils.copyFile(mainJar, out);
184                 return out;
185             }
186         } catch (IOException e) {
187             throw new MojoExecutionException("Cannot copy package file " + files[0].getAbsolutePath()
188                     + " to " + target.getAbsolutePath());
189         }
190     }

192     private void packageApplication() throws MojoExecutionException {
193         String line = getPlay2().getAbsolutePath();
194 
195         CommandLine cmdLine = CommandLine.parse(line);
196         cmdLine.addArgument("package");
197         DefaultExecutor executor = new DefaultExecutor();
198 
199         if (timeout > 0) {
200             ExecuteWatchdog watchdog = new ExecuteWatchdog(timeout);
201             executor.setWatchdog(watchdog);
202         }
203 
204         executor.setWorkingDirectory(project.getBasedir());
205         executor.setExitValue(0);
206         try {
207             executor.execute(cmdLine, getEnvironment());
208         } catch (IOException e) {
209             throw new MojoExecutionException("Error during packaging", e);
210         }
211     }

234     private File moveDistributionArtifactToTarget() throws MojoExecutionException {
235         // The artifact is in dist, this is no more true with play 2.1 as the distribution directory can be customized.
236         // we make the assumption that if dist does not exist, we try in target/dist.
237         File dist = new File(project.getBasedir(), "dist");
238         if (!dist.exists()) {
239             getLog().info("The dist directory does not exist, lookup for the distribution file in target/dist");
240             dist = new File(getBuildDirectory(), "dist");
241             if (!dist.isDirectory()) {
242                 throw new MojoExecutionException("Cannot find the 'dist' directory, " +
243                         "neither 'dist' nor 'target/dist' exist");
244             }
245         }
246         Collection<File> found = FileUtils.listFiles(dist, new String[]{"zip"}, false);
247         if (found.size() == 0) {
248             throw new MojoExecutionException("The distribution file was not found in " + dist.getAbsolutePath());
249         } else if (found.size() > 1) {
250             throw new MojoExecutionException("Too many match for the distribution file in " + dist.getAbsolutePath());
251         }
252 
253         // 1 file
254         File file = found.toArray(new File[0])[0];
255 
256         getLog().info("Distribution file found : " + file.getAbsolutePath());
257 
258         File target = getBuildDirectory();
259         File out = null;
260         if (classifier == null) {
261             out = new File(target, project.getBuild().getFinalName() + ".zip");
262         } else {
263             out = new File(target, project.getBuild().getFinalName() + "-" + classifier + ".zip");
264         }
265 
266         try {
267             getLog().info("Copying " + file.getName() + " to " + out.getName());
268             FileUtils.copyFile(file, out, true);
269         } catch (IOException e) {
270             throw new MojoExecutionException("Can't copy the distribution file to the target folder", e);
271         }
272 
273         // Delete the dist folder if enabled.
274         if (deleteDist) {
275             getLog().debug("Deleting " + dist.getAbsolutePath());
276             FileUtils.deleteQuietly(dist);
277         }
278 
279         return out;
280     }

282     private void moveJavadocAndSourcesArtifactsToTarget() throws MojoExecutionException {
283         File sourceJar = null;
284         File javadocJar = null;
285 
286         File target = getBuildDirectory();
287         File[] files = FileUtils.convertFileCollectionToFileArray(
288                 FileUtils.listFiles(target, new PackageFileFilter(), new PrefixFileFilter("scala-")));
289 
290         for (File file : files) {
291             // We need to select the right file.
292             if (file.getName().endsWith("-sources.jar")) {
293                 sourceJar = file;
294             } else if (file.getName().endsWith("-javadoc.jar")) {
295                 javadocJar = file;
296             }
297         }
298         try {
299             if (sourceJar != null) {
300                 getLog().info("Artifact containing sources found - copying to target");
301                 File out = new File(target, project.getBuild().getFinalName() + "-sources.jar");
302                 FileUtils.copyFile(sourceJar, out);
303             }
304 
305             if (javadocJar != null) {
306                 getLog().info("Artifact containing javadoc found - copying to target");
307                 File out = new File(target, project.getBuild().getFinalName() + "-javadoc.jar");
308                 FileUtils.copyFile(javadocJar, out);
309             }
310         } catch (IOException e) {
311             throw new MojoExecutionException("Can't copy the javadoc and sources file to the target folder", e);
312         }
313     }

315     private void attachArtifactsToProject(File app, File dist) throws MojoExecutionException {
316         Artifact artifact = project.getArtifact();
317 
318         if (StringUtils.isBlank(classifier)) {
319             artifact.setFile(app);
320         } else {
321             projectHelper.attachArtifact(project, "jar", classifier, app);
322         }
323 
324         if (buildDist && attachDist) {
325             projectHelper.attachArtifact(project, "zip", classifier, dist);
326         }
327 
328         if (attachJavadoc) {
329             File javadoc = new File(getBuildDirectory(), project.getBuild().getFinalName() + "-javadoc.jar");
330             if (! javadoc.isFile()) {
331                 throw new MojoExecutionException("Javadoc attachment enabled, but the artifact is not found");
332             }
333             projectHelper.attachArtifact(project, "jar", "javadoc", javadoc);
334         }
335 
336         if (attachSources) {
337             File sources = new File(getBuildDirectory(), project.getBuild().getFinalName() + "-sources.jar");
338             if (! sources.isFile()) {
339                 throw new MojoExecutionException("Sources attachment enabled, but the artifact is not found");
340             }
341             projectHelper.attachArtifact(project, "jar", "sources", sources);
342         }
343

346     private class PackageFileFilter extends AbstractFileFilter {
347 
348         public boolean accept(File dir, String name) {
349             // Must be in the scala dir
350             return dir.getName().startsWith("scala")
351                     // Must ends with Jar
352                     && name.endsWith(".jar");
353         }
